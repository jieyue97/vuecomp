    # 拖拽 组件 gd-drag

    需要实现对 DOM 元素拖拽移动时,可以自定义类 demo

    # 基本用法

    <template>
  <div style="height:200px">
    <GdDrag class="demo" style="height: 50px;width: 50px;background: red;cursor: pointer;"></GdDrag>
  </div>
</template>
<script>
  export default {
    data() {
      return {}
    }
  }
</script>
<style lang="scss" scoped>
  .wrapper {
    height: 300px;
    border: 2px solid #0c5cb3;
  }
  .demo {
    width: 80px;
    height: 80px;
    padding: 0;
    background: red;
  }
</style>
